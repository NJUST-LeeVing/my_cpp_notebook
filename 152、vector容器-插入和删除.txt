#include<iostream>
using namespace std;
#include<vector>

/*vector的插入和删除：
功能描述：对vector容器进行插入、删除操作

函数原型：
push_back(ele);										//尾部插入元素ele
pop_back();											//删除最后一个元素
insert(const_itarator pos, ele);					//迭代器指向位置pos插入元素ele
insert(const_itarator pos, int count, ele);			//迭代器指向位置pos插入count个元素ele
erase(const_iterator start, const_iterator end);	//删除迭代器指向的元素
erase(const_iterator start, const_iterator end);	//删除迭代器从start到end之间的元素
clear();											//删除容器中的所有元素
*/

//打印函数
void PrintVector(vector<int>& v)
{
	for (vector<int>::iterator it = v.begin(); it != v.end(); it++)
	{
		cout << *it << " ";
	}
	cout << endl;
}

//vector容器的插入和删除

void test01()
{
	vector<int>v1;
	//尾插
	v1.push_back(10);
	v1.push_back(20);
	v1.push_back(30);
	PrintVector(v1);

	//尾删
	v1.pop_back();
	PrintVector(v1);

	//插入：指定位置插入数据
	v1.insert(v1.begin(), 100);//第一个参数是迭代器，第二个参数是插入的内容
	PrintVector(v1);
	v1.insert(v1.begin(), 2, 1000);//第一个参数是迭代器，第二个参数是元素个数，第三个是元素内容
	PrintVector(v1);

	//删除:参数是迭代器
	v1.erase(v1.begin());//删除某个内容
	PrintVector(v1);
	v1.erase(v1.begin()+1, v1.end() - 1);//删除某个区间上的内容
	PrintVector(v1);

	//清空：删除所有数据
	v1.clear();
	PrintVector(v1);

}

int main()
{
	test01();
	system("pause");

	return 0;
}
/*
总结：
尾插：push_back();
尾删：pop_back();
插入：insert(位置迭代器);
删除：erase(位置迭代器);
清空：clear();
*/
