#include<iostream>
using namespace std;
#include<string>
//重载关系运算符
class Person
{
public:
	Person(string name,int age)
	{
		m_Name=name;
		m_Age=age;
	}

	//成员函数重载关系运算符==
	bool operator==(Person &p)
	{
		if(this->m_Name ==p.m_Name && this->m_Age ==p.m_Age )
		{
			return true;
		}
		else
		{
			return false;
		}
	}

	string m_Name;
	int m_Age;
};

//全局函数重载关系运算符!=
bool operator!=(Person &p1,Person &p2)
{
	if(p1.m_Name ==p2.m_Name &&p1.m_Name ==p2.m_Name )
	{
		return false;
	}
	else
	{
		return true;
	}
}

void test01()
{
	Person p1("Tom",18);

	Person p2("Lee",18);

	if(p1==p2)
	{
		cout<<"p1和p2是相等的！"<<endl;
	}
	else
	{
		cout<<"p1和p2是不相等的！"<<endl;
	}
	if(p1!=p2)
	{
		cout<<"p1和p2是不相等的！"<<endl;
	}
	else
	{
		cout<<"p1和p2是相等的！"<<endl;
	}

}

int main()
{
	test01();
	system("pause");

	return 0;
}
