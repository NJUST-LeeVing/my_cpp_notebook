#include<iostream>
using namespace std;
#include<set>
#include<string>
/*set容器排序：
学习目标：set容器默认排序规则为从小到大，掌握如何改变排序规则
主要技术点：利用 仿函数 ，可以改变排序规则
*/
/*我也不知道为什么不行*/

class Person
{
public:
	Person(string name, int age)
	{
		this->m_Name = name;
		this->m_Age = age;
	}

	string m_Name;
	int m_Age;

};

//仿函数实际上是一个类
class ComparePerson
{
public:
	bool operator()(const Person& p1, const Person& p2)
	{
		//按照年龄进行降序
		return p1.m_Age > p2.m_Age;
	}
};

void test01()
{

	//对于自定义的数据类型，都会指定排序规则
	set<Person, ComparePerson>s;
	Person p1("李文", 19);
	Person p2("库里",32);
	Person p3("东契奇",20);
	Person p4("詹姆斯",36);

	s.insert(p1);
	s.insert(p2);
	s.insert(p3);
	s.insert(p4);

	for (set<Person, ComparePerson>::iterator it = s.begin(); it != s.end(); it++)
	{
		cout << "姓名：" << it->m_Name << "  年龄：" << it->m_Age << endl;
	}

}


int main()
{
	test01();
	system("pause");

	return 0;

}
