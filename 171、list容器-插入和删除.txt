#include<iostream>
using namespace std;
#include<list>

//打印函数
void PrintList(const list<int>&L)
{
	for (list<int>::const_iterator it = L.begin(); it != L.end(); it++)
	{
		cout << *it << " ";
	}
	cout << endl;
}


void test01()
{
	list<int>L1;
	//尾插
	L1.push_back(10);
	L1.push_back(20);
	L1.push_back(30);
	//头插
	L1.push_front(100);
	L1.push_front(200);
	L1.push_front(300);

	PrintList(L1);

	//尾删
	L1.pop_back();
	PrintList(L1);
	//头删
	L1.pop_front();
	PrintList(L1);

	//insert插入
	list<int>::iterator it = L1.begin();
	L1.insert(++++it, 666);
	PrintList(L1);

	//erase删除
	it = L1.begin();
	L1.erase(++it);
	PrintList(L1);

	//移除
	L1.push_back(10000);
	L1.push_back(10000);
	PrintList(L1);
	L1.remove(10000);//remove可以删除容器中所有与之匹配的值
	PrintList(L1);

	//清空
	L1.clear();
	cout << "清空后：";
	PrintList(L1);
}

int main()
{
	test01();

	system("pause");

	return 0;
}
/*总结：
尾插――push_back
尾删――pop_back
头插――push_front
头删――pop_front
插入――insert
删除――erase
移除――remove
清空――clear
*/
