#include<iostream>
using namespace std;
#include<deque>

//deque赋值操作

void PrintDeque(deque<int>& v)
{
	for (deque<int>::iterator it = v.begin(); it != v.end(); it++)
	{
		cout << *it << " ";
	}
	cout << endl;
}


void test01()
{
	deque<int>d1;
	for (int i = 0; i < 10; i++)
	{
		d1.push_back(i);
	}

	//operator=赋值
	deque<int>d2;
	d2 = d1;
	PrintDeque(d2);

	//assign赋值
	deque<int>d3;
	d3.assign(d2.begin() + 1, d2.end() - 1);
	PrintDeque(d3);

	deque<int>d4;
	d4.assign(10, 100);
	PrintDeque(d4);

}


int main()
{
	test01();

	system("pause");

	return 0;
}
