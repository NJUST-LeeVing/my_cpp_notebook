//指针的作用：指针就是某个内存低地址编号
//指针与内存的区别：指针储存的是内存的地址，变量储存的是内存的值
//须重复观看该节视频内容
//将p理解为指针（地址），*p理解为地址的内容(如：a)
#include<iostream>
using namespace std;

int main()
{
	//1、定义指针
	int a=0;
	//指针定义的语法：   数据类型*指针变量名：
	int* p;//指针的数据类型与变量的数据类型必须一样  //*用于取内容
	//让指针记录变量a的地址(建立指针与变量的关系)
	p=&a;//&为取址符号
	cout<<"a的地址为："<<&a<<endl;
	cout<<"指针p="<<p<<endl;

	//2、使用指针
	//可以通过解引用的方式找到指针指向的内存
	//指针前加*代表解引用，找到指针指向的内存

	*p=1000;
	cout<<"a="<<a<<endl;//指针可修改内存
	cout<<"*p="<<*p<<endl;//通过指针访问内存

	system("pause");
	return 0;
}
